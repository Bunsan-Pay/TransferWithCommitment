1//! `msg.sender` による `transfer`。
2//!
3//! チェーン検証とコントラクト呼び出しに **同一の [`Provider`]** を使います（[`crate::utils::assert_provider_supported_chain`]）。
4//! ブロードキャスト前に [`call`](alloy::contract::SolCallBuilder::call) 相当の **`eth_call` シミュレーション**を行い、失敗時は送信しません（sdk_js の `simulateContract` → `writeContract` と同様）。
5

17/// `transfer(address,address,uint256,bytes32)` — 単一トークン送金。
18///
19/// # 引数
20///
21/// * `provider` — チェーン検証・`TransferWithCommitment` の `.call()`（シミュレーション）と `.send()` に使用。
22/// * `from` — トランザクション送信者（`msg.sender`）。送金元と一致させる必要があります。
23/// * `args` — トークン・送金先・数量・コミットメント。
24///
25/// # 戻り値
26///
27/// ブロードキャストされたトランザクションのハッシュ（[`TxHash`]）。
28///
29/// # エラー
30///
31/// [`crate::SdkError`] — 未設定アドレス、非対応チェーン、RPC／送信失敗など。
32pub async fn transfer<P: Provider>(
33    provider: &P,
34    from: Address,
35    args: SelfSingleTransferArgs,
36) -> Result<TxHash, SdkError> {
37    assert_transfer_contract_configured()?;
38    assert_provider_supported_chain(provider).await?;
39    let c = TransferWithCommitment::new(config::transfer_with_commitment_address(), provider);
40    c.transfer_0(args.token, args.to, args.value, args.commitment)
41        .from(from)
42        .call()
43        .await?;
44    let pending = c
45        .transfer_0(args.token, args.to, args.value, args.commitment)
46        .from(from)
47        .send()
48        .await?;
49    Ok(*pending.tx_hash())
50}

52/// `transfer(TransferDetail[],bytes32)` — ユニファイド明細送金。
53///
54/// # 引数・戻り値・エラー
55///
56/// [`transfer`] と同様。`args` は [`SelfUniCommitTransfersArgs`]。
57pub async fn unified_transfer<P: Provider>(
58    provider: &P,
59    from: Address,
60    args: SelfUniCommitTransfersArgs,
61) -> Result<TxHash, SdkError> {
62    assert_transfer_contract_configured()?;
63    assert_provider_supported_chain(provider).await?;
64    let details: Vec<SolDetail> = args
65        .details
66        .iter()
67        .map(|d| SolDetail {
68            to: d.to,
69            token: d.token,
70            value: d.value,
71        })
72        .collect();
73    let c = TransferWithCommitment::new(config::transfer_with_commitment_address(), provider);
74    c.transfer_1(details.clone(), args.commitment)
75        .from(from)
76        .call()
77        .await?;
78    let pending = c
79        .transfer_1(details, args.commitment)
80        .from(from)
81        .send()
82        .await?;
83    Ok(*pending.tx_hash())
84}

86/// `transfer(CommittedTransferDetail[])` — バッチ送金。
87///
88/// # 引数・戻り値・エラー
89///
90/// [`transfer`] と同様。`args` は [`SelfBatchTransferWithCommitArgs`]。
91pub async fn batch_transfer<P: Provider>(
92    provider: &P,
93    from: Address,
94    args: SelfBatchTransferWithCommitArgs,
95) -> Result<TxHash, SdkError> {
96    assert_transfer_contract_configured()?;
97    assert_provider_supported_chain(provider).await?;
98    let details: Vec<SolCommitted> = args
99        .details
100        .iter()
101        .map(|d| SolCommitted {
102            to: d.to,
103            token: d.token,
104            value: d.value,
105            commitment: d.commitment,
106        })
107        .collect();
108    let c = TransferWithCommitment::new(config::transfer_with_commitment_address(), provider);
109    c.transfer_2(details.clone())
110        .from(from)
111        .call()
112        .await?;
113    let pending = c.transfer_2(details).from(from).send().await?;
114    Ok(*pending.tx_hash())
115}

120    use alloy::providers::ProviderBuilder;

126    #[tokio::test]
127    async fn transfer_fails_when_contract_not_configured() {
128        let provider = ProviderBuilder::new().connect_http("http://127.0.0.1:65531".parse().unwrap());
129        let from = address!("0x1111111111111111111111111111111111111111");
130        let args = SelfSingleTransferArgs {
131            token: from,
132            to: address!("0x3333333333333333333333333333333333333333"),
133            value: U256::from(50u64),
134            commitment: B256::repeat_byte(0xaa),
135        };
136        let r = transfer(&provider, from, args).await;
137        assert!(matches!(r, Err(crate::SdkError::ContractNotConfigured)));
138    }
