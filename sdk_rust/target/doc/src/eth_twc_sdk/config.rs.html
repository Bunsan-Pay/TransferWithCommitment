1//! コントラクトアドレスと対応チェーン ID。
2//!
3//! **デプロイ先アドレスは通常ソース上の定数**です（ランタイムの環境変数差し替え API はありません）。
4//!
5//! `integration-test` フィーチャー有効時のみ、[`transfer_with_commitment_address`] が環境変数 `ETH_TWC_CONTRACT_ADDRESS` を参照します（Anvil 結合テスト用）。
6//!
7//! ## 本番・通常ビルド（管理者向け）
8//!
9//! 通常ビルドでは実アドレスは **非公開の `COMPILE_TIME_TWC`** で決まります。[`transfer_with_commitment_address`] は
10//! `integration-test` 以外ではそれをそのまま返し、[`TRANSFER_WITH_COMMITMENT_ADDRESS`] は同じ値のコンパイル時定数です（関数が定数を読むわけではなく、いずれも同一ソースから派生します）。
11//!
12//! 既定（`test-config` なし）ではゼロアドレスのままです。**運用では `config.rs` 内の `COMPILE_TIME_TWC` の
13//! `not(test-config)` ブランチを、デプロイ済みコントラクトの `address!("0x…")` に差し替えてからビルドしてください。**
14
15#[cfg(feature = "integration-test")]
16use std::sync::OnceLock;
17
18use alloy::primitives::{address, Address};
19
20/// 未設定を表すゼロアドレス。
21pub const ZERO_TRANSFER_ADDRESS: Address = address!("0x0000000000000000000000000000000000000000");
22
23#[cfg(not(feature = "integration-test"))]
24const COMPILE_TIME_TWC: Address = {
25    #[cfg(feature = "test-config")]
26    {
27        address!("0x2222222222222222222222222222222222222222")
28    }
29    #[cfg(not(feature = "test-config"))]
30    {
31        ZERO_TRANSFER_ADDRESS
32    }
33};
34
35#[cfg(feature = "integration-test")]
36static INTEGRATION_TWC: OnceLock<Address> = OnceLock::new();
37
38/// 実行時に解決する `TransferWithCommitment` のデプロイアドレス。
39///
40/// - **通常ビルド** — コンパイル時定数（[`TRANSFER_WITH_COMMITMENT_ADDRESS`] と一致、`integration-test` 以外）。
41/// - **`integration-test`** — 初回アクセス時に `ETH_TWC_CONTRACT_ADDRESS` をパース。未設定・不正ならゼロアドレス扱い。
42#[inline]
43pub fn transfer_with_commitment_address() -> Address {
44    #[cfg(feature = "integration-test")]
45    {
46        *INTEGRATION_TWC.get_or_init(|| {
47            std::env::var("ETH_TWC_CONTRACT_ADDRESS")
48                .ok()
49                .and_then(|s| s.parse().ok())
50                .unwrap_or(ZERO_TRANSFER_ADDRESS)
51        })
52    }
53    #[cfg(not(feature = "integration-test"))]
54    {
55        COMPILE_TIME_TWC
56    }
57}
58
59/// `TransferWithCommitment` コントラクトのアドレス（コンパイル時定数）。
60///
61/// 既定は [`ZERO_TRANSFER_ADDRESS`] です。**ゼロのままでは** [`crate::sign`]・[`crate::verify`]・[`crate::send_transaction`] が失敗します。
62///
63/// `test-config` フィーチャー有効時は単体テスト用の固定アドレスになります。
64///
65/// `integration-test` 有効時は本定数はプレースホルダ（ゼロ）です。実アドレスは [`transfer_with_commitment_address`] を使ってください。
66#[cfg(not(feature = "integration-test"))]
67pub const TRANSFER_WITH_COMMITMENT_ADDRESS: Address = COMPILE_TIME_TWC;
68
69#[cfg(feature = "integration-test")]
70pub const TRANSFER_WITH_COMMITMENT_ADDRESS: Address = ZERO_TRANSFER_ADDRESS;
71
72/// 本 SDK が許可するチェーン ID の一覧。
73///
74/// Ethereum mainnet、Sepolia、Polygon、Arbitrum One。
75/// `integration-test` では Anvil 用に **31337** を追加します。
76#[cfg(not(feature = "integration-test"))]
77pub const SUPPORTED_CHAIN_IDS: &[u64] = &[1, 11155111, 137, 42161];
78
79#[cfg(feature = "integration-test")]
80pub const SUPPORTED_CHAIN_IDS: &[u64] = &[1, 11155111, 137, 42161, 31337];
81
82/// [`transfer_with_commitment_address`] がゼロでないことを検証する。
83///
84/// # 戻り値
85///
86/// - `Ok(())` — コントラクトアドレスが設定済み。
87///
88/// # エラー
89///
90/// - [`crate::SdkError::ContractNotConfigured`] — アドレスが [`ZERO_TRANSFER_ADDRESS`] のまま。
91#[inline]
92pub fn assert_transfer_contract_configured() -> Result<(), crate::SdkError> {
93    if transfer_with_commitment_address() == ZERO_TRANSFER_ADDRESS {
94        return Err(crate::SdkError::ContractNotConfigured);
95    }
96    Ok(())
97}
